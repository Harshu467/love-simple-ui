## Component

import { useState } from "react";
import { Button1 } from "love-simple-ui";

{/* Import CSS modules */}

export const Counter = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <button onClick={() => setCount(count + 1)}>Clicked {count} times</button>
    </div>
  );
};

<Button1>Hello World</Button1>
<Counter />
